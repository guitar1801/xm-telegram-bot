FROM python:3.11-slim

# ติดตั้ง dependencies ที่ Playwright ต้องใช้
RUN apt-get update && apt-get install -y \
    wget \
    libnss3 \
    libatk1.0-0 \
    libatk-bridge2.0-0 \
    libx11-6 \
    libxcb1 \
    libxcomposite1 \
    libxdamage1 \
    libxext6 \
    libxfixes3 \
    libxrandr2 \
    libgbm1 \
    libglib2.0-0 \
    libasound2 \
    libpangocairo-1.0-0 \
    libgtk-3-0 \
    ca-certificates \
    fonts-liberation \
    && rm -rf /var/lib/apt/lists/*

WORKDIR /app

# ก็อปไฟล์ทั้งหมดเข้า Docker
COPY . /app

# ติดตั้ง Python deps
RUN pip install --no-cache-dir -r requirements.txt

# ติดตั้ง Playwright Browsers
RUN playwright install chromium

# ให้ start.sh run ได้
RUN chmod +x /app/start.sh

# รันบอท
CMD ["/app/start.sh"]
